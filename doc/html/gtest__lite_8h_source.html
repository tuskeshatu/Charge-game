465template <typename T>
466bool almostEQ(T a, T b) {
467 // eps: ha a relatív, vagy abszolút hiba ettől kisebb, akkor elfogadjuk
468 T eps = 10 * std::numeric_limits<T>::epsilon(); // 10-szer a legkisebb érték
469 T diff = fabs(a - b);
470 if (diff < eps)
471 return true;
472 T aa = fabs(a);
473 T ba = fabs(b);
474 if (aa < ba) {
475 aa = ba;
476 ba = fabs(a);
477 }
478 return diff < aa * eps;
479}
