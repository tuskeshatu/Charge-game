104Level LevelManager::loadLevel(const std::string &levelName) const
105{
106 // Look for level to be loaded in loadables
107 if (std::find(loadables.begin(), loadables.end(), levelName) != loadables.end())
108 {
109 // Try to open the json file
110 std::ifstream levelFile("./levels/" + levelName + ".json");
111 if (!levelFile)
112 throw std::runtime_error("LevelManager: Level load error: " + levelName + ".json");
113
114 // Deserialize it
115 nlohmann::json jsonData;
116 levelFile >> jsonData;
117
118 // Read name, size, playerStartPos and fill obstacles
119 std::string name = jsonData["name"];
120 sf::Vector2u size(jsonData["size"]["x"], jsonData["size"]["y"]);
121
122 // Create obstacles to be filled
123 std::vector<std::shared_ptr<Obstacle>> obstacles;
124
125 // Auto because type names are confusing with this library
126 // Load fields of each obstacle into obstacle object
127 for (const auto &obstacleData : jsonData["obstacles"])
128 {
129 // Load fields
130 double charge = obstacleData["charge"];
131 sf::Vector2f position(obstacleData["position"]["x"], obstacleData["position"]["y"]);
132 double radius = obstacleData["radius"];
133
134 // Construct obstacle object
135 Obstacle obstacle(radius, charge, position);
136 // Make shared pointer and push to obstacles
137 obstacles.push_back(std::make_shared<Obstacle>(obstacle));
138 }
139
140 // Load playerstartpos
141 sf::Vector2f playerStartPos(jsonData["playerStartPos"]["x"], jsonData["playerStartPos"]["y"]);
142 // Construct level and return it
143 return Level(levelName, size, obstacles, playerStartPos);
144 }
145 // If error occured throw runtime error
146 else
147 {
148 throw std::runtime_error("LevelManager: Level not found: " + levelName + ".json");
149 }
150 if (debug == 5)
151 std::cout << "Loaded level: " + levelName << std::endl;
152}

173// This function saves the given level object to a JSON file.
174void LevelManager::saveLevel(const Level &level)
175{
176 // Open the file for writing
177 std::ofstream levelFile("./levels/" + level.getName() + ".json");
178 if (!levelFile)
179 throw std::runtime_error("LevelManager: Level save error: " + level.getName() + ".json");
180
181 // Write level data to json object
182 nlohmann::json jsonData;
183 jsonData["name"] = level.getName();
184 jsonData["size"]["x"] = level.getSize().x;
185 jsonData["size"]["y"] = level.getSize().y;
186 jsonData["playerStartPos"]["x"] = level.getPlayerStartPos().x;
187 jsonData["playerStartPos"]["y"] = level.getPlayerStartPos().y;
188
189 // Create json objects for every obstacle
190 for (const auto &obstacle : level.getObstacles())
191 {
192 nlohmann::json obstacleData;
193 obstacleData["charge"] = obstacle.get()->getElectricCharge();
194 obstacleData["position"]["x"] = obstacle.get()->getBody()->getPosition().x;
195 obstacleData["position"]["y"] = obstacle.get()->getBody()->getPosition().y;
196 obstacleData["radius"] = obstacle.get()->getBody()->getRadius();
197
198 jsonData["obstacles"].push_back(obstacleData);
199 }
200
201 // Write to file
202 levelFile << jsonData;
203
204 // Add level name to loadables if it is not already present
205 if (std::find(loadables.begin(), loadables.end(), level.getName()) == loadables.end())
206 loadables.push_back(level.getName());
207
208 if (debug == 5)
209 std::cout << "Saved level: " + level.getName() << std::endl;
210
211 levelFile.close();
212}
