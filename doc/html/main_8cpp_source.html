159void handleGameEvent()
160{
161 // Create event object and poll for events
162 sf::Event evnt;
163 while (window.pollEvent(evnt))
164 {
165 switch (evnt.type)
166 {
167 // Window is closed, vectors are cleared, program terminates
168 case sf::Event::Closed:
169 level.clearObstacles();
170 gameDrawables.clear();
171 menuDrawables.clear();
172 window.close();
173 exit(0);
174 break;
175 // Window is resized
176 case sf::Event::Resized:
177 resizeView(evnt);
178 break;
179 // If key press occured (event is used for detecting key presses to avoid sticky keys)
180 case sf::Event::KeyPressed:
181 // Escape pauses
182 if (evnt.key.code == sf::Keyboard::Escape)
183 isPause = !isPause;
184 // R resets based on modifyer keys
185 if (evnt.key.code == sf::Keyboard::R)
186 {
187 // LCtrl + R: If in editor mode, this allows the player to not reset level, but set start speed
188 if (isEditorMode && sf::Keyboard::isKeyPressed(sf::Keyboard::LControl))
189 {
190 setStartSpeed();
191 runGame();
192 }
193 // In normal case (not in editor mode and if in editor mode only if R is pressed) just resets game.
194 startGame();
195 }
196
197 default:
198 break;
199 }
200 }
201}

246void handleEditorModeInput()
247{
248 // Input is discarded if window is not in focus
249 if (!window.hasFocus())
250 return;
251
252 // Key bindings:
253 // Space: places the player at current mouse cursor position
254 if (sf::Keyboard::isKeyPressed(sf::Keyboard::Space))
255 {
256 sf::Vector2f zeroSpeed(0.0f, 0.0f);
257 player.setSpeed(zeroSpeed);
258 sf::Vector2i mousePos = sf::Mouse::getPosition(window);
259 sf::Vector2f mousePosFloat(static_cast<float>(mousePos.x), static_cast<float>(mousePos.y));
260 player.setPosition(mousePosFloat);
261 }
262 // Z: zeroes player speed
263 if (sf::Keyboard::isKeyPressed(sf::Keyboard::Z))
264 {
265 sf::Vector2f zeroSpeed(0.0f, 0.0f);
266 player.setSpeed(zeroSpeed);
267 }
268 // Left click:
269 if (sf::Mouse::isButtonPressed(sf::Mouse::Left))
270 {
271 // Get mouse position when click happens
272 sf::Vector2f mousePos;
273 mousePos.x = sf::Mouse::getPosition(window).x;
274 mousePos.y = sf::Mouse::getPosition(window).y;
275 // If LCtrl modifier key is pressed: places negative charge
276 if (sf::Keyboard::isKeyPressed(sf::Keyboard::LControl))
277 {
278 // Create obstacle and add to level, push to drawables.
279 std::shared_ptr<Obstacle> newObstacle (std::make_shared<Obstacle>(7.0f, -1500.0f, mousePos));
280 level.addObstacle(newObstacle);
281
282 gameDrawables.push_back(newObstacle->getBody());
283 }
284 // If LAlt modifier key is pressed: removes obstacles the mouse touches
285 else if (sf::Keyboard::isKeyPressed(sf::Keyboard::LAlt))
286 {
287 // Check for each obstacle if mouse is touching
288 for (size_t i = 0; i < level.getObstacles().size(); i++)

290 // If touching, remove from obstacles and drawables
291 if (level.getObstacles()[i]->getBody()->getGlobalBounds().contains(mousePos.x, mousePos.y))
292 {
293 level.removeObstacle(i);
294 gameDrawables.erase(gameDrawables.begin() + i + 1);
295 }
296 }
297 }
298 }
299 // Right click: If LCtrl modifier key is pressed places positive charge
300 if (sf::Mouse::isButtonPressed(sf::Mouse::Right) && sf::Keyboard::isKeyPressed(sf::Keyboard::LControl))
301 {
302 // Get mouse position when click happens
303 sf::Vector2f mousePos;
304 mousePos.x = sf::Mouse::getPosition(window).x;
305 mousePos.y = sf::Mouse::getPosition(window).y;
306
307 // Create obstacle and add to level, push to drawables.
308 std::shared_ptr<Obstacle> newObstacle (std::make_shared<Obstacle>(7.0f, 1500.0f, mousePos));
309 level.addObstacle(newObstacle);
310 gameDrawables.push_back(newObstacle->getBody());

354void setStartSpeed()
355{
356 // Construct isMouseOnPlayer with the vector from if the vector from the mouse to the player is shorter than the radius of the player
357 // To avoid calculating vectors, check if the x and y coordinates of the mouse is inside the circle
358 bool isMouseOnPlayer = std::abs(sf::Mouse::getPosition(window).x - player.getBody()->getPosition().x) < player.getBody()->getRadius() && std::abs(sf::Mouse::getPosition(window).y - player.getBody()->getPosition().y) < player.getBody()->getRadius();
359
360 // Wait for mouse click, when mouse is over player
361 while (!(sf::Mouse::isButtonPressed((sf::Mouse::Left)) && isMouseOnPlayer) && window.isOpen())
362 {
363 // Handle events for allowing closing and pausing
364 handleGameEvent();
365
366 // Handle editor inputs if editor mode is enabled
367 if (isEditorMode)
368 handleEditorModeInput();
369
370 render();
371
372 // If paused dont calculate anything
373 if (isPause)
374 displayPauseOverlay();
375 else
376 isMouseOnPlayer = std::abs(sf::Mouse::getPosition(window).x - player.getBody()->getPosition().x) < player.getBody()->getRadius() && std::abs(sf::Mouse::getPosition(window).y - player.getBody()->getPosition().y) < player.getBody()->getRadius();
377 }
378
379 // If clicked on player initialize starting speed and create arrow rectangle
380 sf::Vector2f startSpeed(0.0f, 0.0f);
381 std::shared_ptr<sf::RectangleShape> arrow(std::make_shared<sf::RectangleShape>(sf::Vector2f(0.0f, arrowWidth)));
382 arrow->setFillColor(sf::Color::Magenta);
383 arrow->setOrigin(0.0f, arrowWidth / 2);
384 gameDrawables.push_back(arrow);
385
386 // Do until mouse is pressed (the player is dragging)
387 while (sf::Mouse::isButtonPressed((sf::Mouse::Left)))
388 {
389 // Handle events
390 handleGameEvent();
391 if (isEditorMode)
392 handleEditorModeInput();
393
394 render();
395
396 if (isPause)
397 displayPauseOverlay();
398 // If not paused draw arrow accordingly
399 else
400 {
401 // Calculate the size the arrow needs to be by calculating the length of the vector stretching from the mouse to the player
402 arrow->setPosition(player.getBody()->getPosition());
403 arrow->setSize(sf::Vector2f(std::sqrt(startSpeed.x * startSpeed.x + startSpeed.y * startSpeed.y), arrowWidth));
404
405 // Calculate rotation angle by calculating atan from the triangle the mouse and the player is drawing out and apply rotation
406 float angle = std::atan2((sf::Mouse::getPosition(window).y - player.getBody()->getPosition().y), (sf::Mouse::getPosition(window).x - player.getBody()->getPosition().x)) * 180.0f / M_PI;
407 arrow->setRotation(angle);
408
409 // Set start speed by basically factoring by the length of the vector stretching between the player and mouse
410 startSpeed.x = (player.getBody()->getPosition().x - sf::Mouse::getPosition(window).x);
411 startSpeed.y = (player.getBody()->getPosition().y - sf::Mouse::getPosition(window).y);
412 if (debug == 4)
413 std::cout << "Start speed:\t" << std::sqrt(startSpeed.x * startSpeed.x + startSpeed.y * startSpeed.y)
414 << "\t\tx: " << startSpeed.x << "\ty: " << startSpeed.y << std::endl;
415 }
416 }
417
418 // Remove arrow from drawables and set speed to calculated starting speed
419 gameDrawables.pop_back();
420 player.setSpeed(startSpeed);
421}

423// Run method with game loop
432void runGame()
433{
434 // Restart game clock measuring deltaTime between iterations of simulation cycles
435 gameClock.restart();
436
437 // Game loop
438 while (window.isOpen())
439 {
440 // If window loses focus, pause is requested
441 if (!window.hasFocus())
442 isPause = true;
443
444 // If pause is requested, display pause overlay
445 if (isPause)
446 displayPauseOverlay();
447
448 // Set deltaTime
449 deltaTime = gameClock.restart().asSeconds();
450
451 // Handle events
452 handleGameEvent();
453
454 // Handle editor inputs if editor mode is enabled
455 if (isEditorMode)
456 handleEditorModeInput();
457
458 // Run iteration of physics simulation
459 physics.updatePlayer();
460
461 // Render drawables
462 render();
463 }
464}

474{

523 deleteLevelOption->setFillColor(sf::Color::Red);

526 deleteLevelOption->setPosition(30.0f, window.getSize().y - 30.0f);

534 editorModeOption->setPosition(window.getSize().x - 30.0f, window.getSize().y - 30.0f);

561 menuItem->setOrigin(menuItem->getGlobalBounds().width / 2.0f, menuItem->getGlobalBounds().height / 2.0f);

564 menuItems.push_back(menuItem);

568 for (const std::shared_ptr<sf::Text> &menuItem : menuItems)

573 // Menu loop for
574 bool isDeleteMode = false;
575 while (window.isOpen())

578 if (debug == 6)

585 // Polling for events (for menu item clicks and other resize and close events)
586 while (window.pollEvent(evnt))

590 level.clearObstacles();
591 gameDrawables.clear();
592 menuDrawables.clear();
593 window.close();

601 deleteLevelOption->setPosition(30.0f, window.getSize().y - 30.0f);

882 break;

885 }

889 inputText->setString("Enter level name: " + levelName);
