25// Calculate electric force
26const sf::Vector2f PhysicsEngine::calculateElectricForce() const
27{
28
29 // Calculate force vector for each obstacle with the player and sum them
30 // Formula: F(r) = k * q_player sum(q_obstacle * (ri / abs(ri)^3))
31 // Where ri is a vector pointing from the obstacle to the player
32 sf::Vector2f totalForce(0.0, 0.0);
33
34 // Calculate force vector for each obstacle with the player and sum them
35 // Formula: F(r) = k * q_player sum(q_obstacle * (ri / abs(ri)^3))
36 // Where ri is a vector pointing from the obstacle to the player
37 for (const std::shared_ptr<Obstacle> &obstacle : level.getObstacles())
38 {
39 // r vector pointing from the obstacle to the player
40 sf::Vector2f ri = player.getBody()->getPosition() - obstacle.get()->getBody()->getPosition();
41 // Devide ri by 1000 to get realistic distances
42 ri.x /= 1000;
43 ri.y /= 1000;
44 double riLengthSquared = ri.x * ri.x + ri.y * ri.y;
45
46 // It is divided by the cube of riLength
47 // x component
48 totalForce.x += obstacle.get()->getElectricCharge() * (ri.x / (riLengthSquared * std::sqrt(riLengthSquared)));
49 // y component
50 totalForce.y += obstacle.get()->getElectricCharge() * (ri.y / (riLengthSquared * std::sqrt(riLengthSquared)));
51 }
52 // Multiply the sum to get total force
53 totalForce.x *= k * player.getElectricCharge();
54 totalForce.y *= k * player.getElectricCharge();
55
56 return totalForce;
57}

62 // Friction is linearly proportional to the speed of the player (similar to drag irl)

67// Checks for collisions between Player and walls and obstacles.
68void PhysicsEngine::checkCollision()
69{
70 // Get player position
71 const sf::Vector2f playerPos(player.getBody()->getPosition());
72 // Check for each obstacle
73 for (const std::shared_ptr<Obstacle> &obstacle : level.getObstacles())
74 {
75 sf::Vector2f obstaclePos(obstacle.get()->getBody()->getPosition());
76 // If the length of the vector between the obstacle and the player is shorter than the radius they are colliding
77 // To avoid calculating sqrt basically: x^2+y^2 <= (r1 + r2)^2
78 if ((playerPos.x - obstaclePos.x) * (playerPos.x - obstaclePos.x) + (playerPos.y - obstaclePos.y) * (playerPos.y - obstaclePos.y) <= (player.getBody()->getRadius() + obstacle.get()->getBody()->getRadius()) * (player.getBody()->getRadius() + obstacle.get()->getBody()->getRadius()))
79 isPause = true;
80 }
81
82 // Check collision with walls of the window, simulate perfectly elastic collision, where walls have infinite weight
83 // So just set the corresponding component of player's speed to its opposite
84 sf::Vector2f windowSize(window.getSize());
85 sf::Vector2f playerSpeed(player.getSpeed());
86 if (playerPos.x < 0 || playerPos.x > windowSize.x)
87 playerSpeed.x = -playerSpeed.x;
88 if (playerPos.y < 0 || playerPos.y > windowSize.y)
89 playerSpeed.y = -playerSpeed.y;
90
91 // And finally set the speed
92 player.setSpeed(playerSpeed);
93}

96void PhysicsEngine::updatePlayer()
97{
98 // Initialize total force
99 sf::Vector2f totalForce(0.0, 0.0);
100
101 // Add the electric part
102 totalForce += calculateElectricForce();
103
104 // Subtract a friction force proportionally linked to the speed
105 totalForce -= calculateFrictionForce();
106
107 if (debug == 2)
108 std::cout << "total force:\t" << std::sqrt(totalForce.x * totalForce.x + totalForce.y * totalForce.y)
109 << "\t\tx: " << totalForce.x << "\ty: " << totalForce.y << std::endl;
110
111 // To get acceleration divide force by mass
112 sf::Vector2f acceleration(0.0, 0.0);
113 acceleration.x = totalForce.x / player.getMass();
114 acceleration.y = totalForce.y / player.getMass();
115
116 // Update player movement
117 player.updateMovement(acceleration);
118
119 // And check for collisions
120 checkCollision();
121}
